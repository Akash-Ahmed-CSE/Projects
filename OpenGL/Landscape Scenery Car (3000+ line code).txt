int frameNumber = 0;

// Disk Draw for Wheel
void drawDisk(double radius) {
	int d;
	glBegin(GL_POLYGON);
	for (d = 0; d < 32; d++) {
		double angle = 2 * PI / 32 * d;
		glVertex2d( radius*cos(angle), radius*sin(angle));
	}
	glEnd();
}

// Wheel of Car for rotating while moving
void drawWheel() {

	glColor3f(0,0,0);
	drawDisk(1);
	glColor3f(0.75f, 0.75f, 0.75f);
	drawDisk(0.8);
	glColor3f(0,0,0);
	drawDisk(0.2);
	glRotatef(frameNumber*20,0,0,1);

	glEnd();
}

// Upper layer of Car with Moving Forward and Backward
void drawCart(float a, float color) {
	glPushMatrix();
	glTranslatef(-1.5f+a, -0.1f, 0);
	glScalef(0.8f,0.8f,1);
	drawWheel();
	glPopMatrix();
	glPushMatrix();
	glTranslatef(1.5f+a, -0.1f, 0);
	glScalef(0.8f,0.8f,1);
	drawWheel();
	glPopMatrix();
	glColor3f(1,0+color,0);
	glBegin(GL_POLYGON);
	glVertex2f(-2.88f+a,1);
	glVertex2f(-2.5f+a,0);
	glVertex2f(2.5f+a,0);
	glVertex2f(2.88f+a,1);
	glVertex2f(2.88f+a,1);
	glVertex2f(2.5f+a,2);
	glVertex2f(-2.5f+a,2);
	glEnd();
}


// Car Function
void car(float a,float color){
    glPushMatrix();
    glScaled(0.01,0.01,1);
    drawCart(a,color);
    glPopMatrix();
}

// Framing Function for Car
void doFrame(int v) {
    frameNumber++;
    glutPostRedisplay();
    glutTimerFunc(30,doFrame,0);
}

// Font Select for Cover Page
void setFont(void *font){
	currentfont = font;
}

// Setup to Draw String
void drawstring(float x, float y, float z, char *string){
	char *c;
	glRasterPos3f (x, y, z);

	for(c = string; *c != '\0'; c++){
        glColor3f (0.0, 0.0, 0.0);
		glutBitmapCharacter (currentfont, *c);
	}
}

// Multi-discipline Display as Medium
void mydisplay(void){
	if(flag==0)
		frontscreen();
	if(flag==1)
		display();
        glutSwapBuffers();
}

// Keyboard Command for Display
void myKeyboard(unsigned char key, int x, int y){
    switch(key){
        case 13:
            if(flag == 0)
            {
                flag=1;
                display();
            }
            break;
        default:
            break;
    }
}

// Initialize Open GL
void init (void){
    glClearColor (0.0, 0.0, 0.0, 0.0);
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    glOrtho(0.0, 1.0, 0.0, 1.0, -1.0, 1.0);
}

    glutInit(&argc, argv);
    glutInitDisplayMode (GLUT_DOUBLE | GLUT_RGB);
    glutInitWindowSize (920, 600);
    glutInitWindowPosition (220, 220);
    glutCreateWindow ("Rahman, M. M.");
    init ();
    glutDisplayFunc(mydisplay);
	glutKeyboardFunc(myKeyboard);
	glutTimerFunc(200,doFrame,0);
    glutMainLoop();

    return 0;
}
